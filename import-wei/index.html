Try It

wei.resolve(name)

Resolve a .wei name to an address

wei.reverseResolve(address)

Get the .wei name for an address

wei.isWei(string)

Check if string is a .wei name

API Reference

// Resolve name to address
const addr = await wei.resolve('name.wei')
// => '0x...' or null

// Reverse resolve address to name
const name = await wei.reverseResolve('0x...')
// => 'name.wei' or null

// Smart resolve (addresses pass through)
const addr = await wei.resolveAny(input)

// Check if .wei name
wei.isWei('name.wei')  // true
wei.isWei('0x...')     // false

// Custom RPC endpoint
wei.config({ rpc: 'https://eth-mainnet.g.alchemy.com/v2/KEY' })

// Bridge ETH from Mainnet to Base (requires ethers.js signer)
await wei.bridgeToBase('name.wei', '0.1', signer)
await wei.bridgeToBase(null, '0.1', signer)  // bridge to self

// Contract address
wei.CONTRACT  // '0x0000000000696760E15f265e828DB644A0c242EB'
wei.BASE_PORTAL // '0x49048044D57e1C92A77f79988d21Fa8fAF74E97e'

Example: Payment Form

<input id="to" placeholder="0x... or name.wei">

<script src="https://import.wei.domains/wei.min.js"></script>
<script>
async function send() {
  const input = document.getElementById('to').value
  const address = await wei.resolveAny(input)

  if (!address) return alert('Invalid recipient')

  // Send transaction to resolved address
  await ethereum.request({
    method: 'eth_sendTransaction',
    params: [{ to: address, value: '0x...' }]
  })
}
</script>
